def login_view(request): if request.method == 'POST': username = request.POST['username'] password = request.POST['password'] user = authenticate(request, username=username, password=password) if user: login(request, user) return redirect('todo_list') else: return render(request, 'todos/login.html', {'error': 'Неверный логин или пароль'}) return render(request, 'todos/login.html')
